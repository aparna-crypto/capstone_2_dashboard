pipeline {
    agent any
    
    environment{
        dockerhub=credentials('docker')
    }
    stages {
        stage('Clone repo') {
            steps {
                checkout scm
            }
        }
    
        stage('Build Image'){
            steps {
                bat 'docker build -t aparnaramesh12/dashboard .'
            }
        }
        
        stage('Push to DockerHub') {
            steps {
                bat 'echo $dockerhub_PSW | docker login -u $dockerhub_USR --password-stdin'
                bat 'docker push aparnaramesh12/dashboard'
            }
        }
  
        stage('Run Image') {
            steps {
                bat 'docker run -p 8051:8051 capstone/dashboard'
            }
        }

        stage('Testing') {
            steps {
                echo 'Process Complete'
            } 
        }
    }
}
