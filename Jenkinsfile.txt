pipeline {
    agent any
    
    environment{
        DockerHub=credentials('DockerHub')
    }
    stages {
        stage('Clone repo') {
            steps {
                checkout scm
            }
        }
    
        stage('Build Image'){
            steps {
                sh 'sudo docker build -t capstone/dashboard .'
            }
        }
        
        stage('Push to DockerHub') {
            steps {
                sh 'docker push capstone/dashboard'
            }
        }
  
        stage('Run Image') {
            steps {
                sh 'docker run -p 8051:8051 capstone/dashboard'
            }
        }

        stage('Testing') {
            steps {
                echo 'Process Complete'
            }
        }
    }
}
