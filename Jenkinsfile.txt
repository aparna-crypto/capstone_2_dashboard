pipeline {
    agent any
    
    environment{
        dockerhub=credentials('docker')
    }
    stages {
        stage('Clone repo') {
            steps {
                checkout scm
            }
        }
    
        stage('Build Image'){
            steps {
                bat 'docker build -t cap/dashboard .'
            }
        }
  

        stage('Testing') {
            steps {
            
                bat 'pytest test_sample.py'
                echo 'Process Complete'
            } 
        }
    }
}
