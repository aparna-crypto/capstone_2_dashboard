pipeline {
    agent any
    
    environment{
        dockerhub=credentials('docker')
    }
    stages {
        stage('Clone repo') {
            steps {
                checkout scm
            }
        }
    
        stage('Build Image'){
            steps {
                bat 'docker build -t cap/dashboard .'
            }
        }
  
        stage('Run Image') {
            steps {
                bat 'docker run -p 8051:8051 cap/dashboard'
            }
        }

        stage('Testing') {
            steps {
                echo 'Process Complete'
            } 
        }
    }
}
