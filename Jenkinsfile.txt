pipeline {
    agent any

    stages {
        stage('Clone repo') {
            steps {
                checkout scm
            }
        }
    
        stage('Build Image'){
            steps {
                sh 'sudo docker build -t dashboard .'
            }
        }
  
        stage('Run Image') {
            steps {
                sh 'docker run -p 8051:8051 dashboard'
            }
        }

        stage('Testing') {
            steps {
                echo 'Process Complete'
            }
        }
    }
}
